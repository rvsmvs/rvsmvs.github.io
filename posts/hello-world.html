def classify_artifact(value: str) -> str:
    if value.startswith(('http://', 'https://')):
        return 'URL'
    if value.count('.') == 3 and all(p.isdigit() and 0 <= int(p) <= 255 for p in value.split('.')):
        return 'IPv4'
    return 'Unknown'

print(classify_artifact('192.168.1.10'))
